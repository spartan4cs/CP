bool Solution::hotel(vector &arrive, vector &depart, int K) {
int size=arrive.size();
sort(arrive.begin(), arrive.end());
sort(depart.begin(), depart.end());
int c=0, i=0, j=0;
while(i<size && j<size){
if(arrive[i]>=depart[j]){
c++;
j++;
i++;
}
else{
i++;
}
}
// cout<<size-c;
return (size-c)<=K;
}
